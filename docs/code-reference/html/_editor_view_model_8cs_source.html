249 private async Task Add(ITask task)
250 {
251 var taskComposite = TaskListView as ITaskComposite;
252 if (SelectedTasks.Count == 1)
253 {
254 taskComposite = (ITaskComposite)SelectedTasks.First();
255 SelectedTasks.Clear();
256 }
257 var result = await AddModal(_addTasksDialog, task);

269 [ReactiveCommand(CanExecute = nameof(_canExecuteEdit))]
270 private async Task Edit()
271 {
272 var item = SelectedTasks.First();
273 SelectedTasks.Clear();
274 var editorService = (IEditorService)null;
275 if (item is ITaskElement taskElement)
276 {
277 var taskElementsEditorService = _taskElementsEditorProxy;
278 taskElementsEditorService.Target = taskElement;
279 editorService = taskElementsEditorService;
280 }
281 else
282 {
283 var tasksEditorService = _tasksEditorProxy;
284 tasksEditorService.Target = item;
285 editorService = tasksEditorService;
286 }
287 var result = await AddModal(_editTaskDialog, editorService);
288 if (result)
289 {
290 editorService.ApplyChanges();
291 _session.EditTask(item);
292 }
293 }

316 [ReactiveCommand(CanExecute = nameof(_canExecuteCopy))]
317 private async Task Copy()
318 {
319 var items = SelectedTasks.ToList();
320 SelectedTasks.Clear();
321 var result = await AddModal(_copyTasksDialog,
322 new ItemsTasksViewModelArgs(items, TaskListView, _session.Tasks));
323 if (result != null)
324 {
325 var copyList = new List<ITask>();
326 foreach (var task in items)
327 {
328 if (task is ICloneable cloneable)
329 {
330 for (var i = 0; i < result.Count; ++i)
331 {
332 copyList.Add((ITask)cloneable.Clone());
333 }
334 }
335 }
336 _session.AddTasks(copyList, result.List as ITaskComposite);
337 }
338 }

344 [ReactiveCommand(CanExecute = nameof(_canExecuteSwap))]
345 private void Swap()
346 {
347 var items = SelectedTasks.ToList();
348 SelectedTasks.Clear();
349
350 var item1 = items.First();
351 var item2 = items.Last();
352
353 var list = item1.ParentTask ?? _session.Tasks as IList<ITask>;
354
355 var index1 = list.IndexOf(item1);
356 var index2 = list.IndexOf(item2);
357
358 list[index1] = item2;
359 list[index2] = item1;
360 }
361 }
