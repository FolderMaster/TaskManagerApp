148 [ReactiveCommand]
149 private void Update()
150 {
151 if (_session.Tasks == null)
152 {
153 return;
154 }
155 CalendarIntervals.Clear();
156 var tasks = TaskHelper.GetTaskElements(_session.Tasks);
157 foreach (var task in tasks)
158 {
159 foreach(var timeInterval in task.TimeIntervals)
160 {
161 CalendarIntervals.Add(new CalendarInterval(timeInterval, task));
162 }
163 }
164 _tasksSchedulerDictionary = tasks.SelectMany(t => t.TimeIntervals).GroupBy(i => i.Start).
165 Where(g => g.Key > DateTime.Now).ToDictionary(g => g.Key,
166 g => tasks.Where(task => task.TimeIntervals.Any(i => i.Start == g.Key)));
167 _timeScheduler.Timepoints.Clear();
168 _timeScheduler.Timepoints.AddRange(_tasksSchedulerDictionary.Keys);
169 }
