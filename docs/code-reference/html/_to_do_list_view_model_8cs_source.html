109 [ReactiveCommand]
110 public void Update()
111 {
112 if (_session.Tasks == null)
113 {
114 return;
115 }
116 var tasks = TaskHelper.GetTaskElements(_session.Tasks.ToList());
117 var uncompletedTasks = tasks.Where(t => !TaskHelper.IsTaskCompleted(t));
118 var toDoList = uncompletedTasks.Select(CreateToDoListElement);
119 if (IsLaggingFilter)
120 {
121 toDoList = toDoList.Where(e => e.IsLagging);
122 }
123 if (IsExpiredFilter)
124 {
125 toDoList = toDoList.Where(e => e.IsExpired);
126 }
127 if (IsTimeSort)
128 {
129 toDoList = toDoList.OrderBy(e =>
130 e.TaskElement.PlannedTime - e.TaskElement.SpentTime);
131 }
132 if (IsRealSort)
133 {
134 toDoList = toDoList.OrderBy(e =>
135 e.TaskElement.PlannedReal - e.TaskElement.ExecutedReal);
136 }
137 if (IsExecutionChanceSort)
138 {
139 toDoList = toDoList.OrderBy(e => e.ExecutionChance);
140 }
141 if (IsDifficultSort)
142 {
143 toDoList = toDoList.OrderBy(e => e.TaskElement.Difficult);
144 }
145 if (IsPrioritySort)
146 {
147 toDoList = toDoList.OrderBy(e => e.TaskElement.Priority);
148 }
149 ToDoList = toDoList;
150 }

152 private ToDoListElement CreateToDoListElement(ITaskElement taskElement)
153 {
154 var executionChance = _progressLearningController.IsValidModel ?
155 (double?)_progressLearningController.Predict(taskElement) : null;
156 var isExpired = taskElement.Deadline < DateTime.Now;
157 var isLagging = !isExpired && taskElement.Deadline < DateTime.Now +
158 (taskElement.PlannedTime - taskElement.SpentTime);
159
160 return new ToDoListElement(taskElement, executionChance, isLagging, isExpired);
