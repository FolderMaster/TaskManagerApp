115 private void UpdateTasksCountStatistics()
116 {
117 if (_session.Tasks == null)
118 {
119 return;
120 }
121 var tasks = TaskHelper.GetTaskElements(_session.Tasks);
122 var uncompletedTasks = tasks.Where(t => !TaskHelper.IsTaskCompleted(t));
123
124 var difficultDiagramContent = _resourceService.GetResource("DifficultDiagramContent");
125 var priorityDiagramContent = _resourceService.GetResource("PriorityDiagramContent");
126
127 UncompletedTasksCountByCategoryStatistic = uncompletedTasks.
128 GroupBy(t => ((TaskMetadata)t.Metadata).Category).
129 Select(g => new StatisticElement(g.Count(), $"{g.Key}"));
130 UncompletedTasksCountByTagsStatistic = uncompletedTasks.
131 SelectMany(t => ((TaskMetadata)t.Metadata).Tags, (task, tag) =>
132 new { Task = task, Tag = tag }).GroupBy(e => e.Tag).
133 Select(g => new StatisticElement(g.Count(), $"{g.Key}"));
134 UncompletedTasksCountByPriorityStatistic = uncompletedTasks.GroupBy(t => t.Priority).
135 Select(g => new StatisticElement(g.Count(), $"{priorityDiagramContent} {g.Key}"));
136 UncompletedTasksCountByDifficultStatistic = uncompletedTasks.GroupBy(t => t.Difficult).
137 Select(g => new StatisticElement(g.Count(), $"{difficultDiagramContent} {g.Key}"));
138 }

143 private void UpdateExpiredTasksStatistics()
144 {
145 if (_session.Tasks == null)
146 {
147 return;
148 }
149 var tasks = TaskHelper.GetTaskElements(_session.Tasks);
150 var expiredTasks = tasks.Where(t => !TaskHelper.HasTaskExpired(t) &&
151 TaskHelper.HasTaskExpired(t, SelectedTime));
152
153 var count = expiredTasks.Count();
154 var plannedTime = expiredTasks.Aggregate(TimeSpan.Zero,
155 (sum, task) => sum + task.PlannedTime).Hours;
156 var spentTime = expiredTasks.Aggregate(TimeSpan.Zero,
157 (sum, task) => sum + task.SpentTime).Hours;
158
159 var countDiagramContent = _resourceService.GetResource("CountDiagramContent");
160 var plannedTimeDiagramContent = _resourceService.GetResource("PlannedTimeDiagramContent");
161 var spentTimeDiagramContent = _resourceService.GetResource("SpentTimeDiagramContent");
162
163 ExpiredTasksStatistic =
164 [
165 new StatisticElement(count, countDiagramContent?.ToString()),
166 new StatisticElement(plannedTime, plannedTimeDiagramContent?.ToString()),
167 new StatisticElement(spentTime, spentTimeDiagramContent?.ToString())
168 ];
169 }
170
174 private void UpdateTimeTasksStatistic()
175 {
176 if (_session.Tasks == null)
177 {
178 return;
179 }
180 var tasks = TaskHelper.GetTaskElements(_session.Tasks);
181 var uncompletedTasks = tasks.Where(t => !TaskHelper.IsTaskCompleted(t));
182
183 var plannedTime = uncompletedTasks.Aggregate(TimeSpan.Zero,
184 (sum, task) => sum + task.TimeIntervals.Duration).Hours;
185 var unplannedTime = uncompletedTasks.Aggregate(TimeSpan.Zero,
186 (sum, task) => sum + task.PlannedTime).Hours - plannedTime;
187
188 var plannedTimeDiagramContent = _resourceService.GetResource("PlannedTimeDiagramContent");
189 var unplannedTimeDiagramContent = _resourceService.GetResource("UnplannedTimeDiagramContent");
190
191 TasksTimeStatistic =
192 [
193 new StatisticElement(plannedTime, plannedTimeDiagramContent?.ToString()),
194 new StatisticElement(unplannedTime, unplannedTimeDiagramContent?.ToString())
195 ];
196 }
